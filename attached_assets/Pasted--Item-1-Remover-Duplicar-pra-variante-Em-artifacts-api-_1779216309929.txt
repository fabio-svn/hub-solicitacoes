c) **Função toggle** — adicionar no script global (perto de outras 
   funções públicas do editor):

```js
function toggleGroup(key, titleEl) {
  const body = document.getElementById(key);
  if (!body) return;
  const isExpanded = body.classList.contains('expanded');
  if (isExpanded) {
    body.classList.remove('expanded');
    titleEl.classList.remove('expanded');
  } else {
    body.classList.add('expanded');
    titleEl.classList.add('expanded');
  }
  // Persistir estado em sessionStorage
  try {
    const states = JSON.parse(sessionStorage.getItem('templateGroupStates') || '{}');
    states[key] = !isExpanded;
    sessionStorage.setItem('templateGroupStates', JSON.stringify(states));
  } catch {}
}

function restoreGroupStates() {
  try {
    const states = JSON.parse(sessionStorage.getItem('templateGroupStates') || '{}');
    Object.entries(states).forEach(([key, expanded]) => {
      if (expanded) {
        const body = document.getElementById(key);
        const title = body?.previousElementSibling;
        if (body && title) {
          body.classList.add('expanded');
          title.classList.add('expanded');
        }
      }
    });
  } catch {}
}
```

d) **Chamar restoreGroupStates** após o render da lista. Procurar onde 
`renderListView` ou similar é chamado e adicionar `restoreGroupStates()` 
no final, ou dentro da função após `el.innerHTML = html`.

e) **Default = recolhido**: como CSS tem `max-height: 0` e a classe 
`expanded` só vem do sessionStorage (vazio na primeira visita), todos 
começam recolhidos. ✓
