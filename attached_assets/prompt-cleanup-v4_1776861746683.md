## 1. BUG — `goTo()` não desabilita os botões durante a verificação de auth

```js
// Problema: se o usuário clicar nos dois botões rapidamente antes do
// Auth.init() responder, podem ser disparadas duas navegações simultâneas.
// Também sem feedback visual de loading durante o await.

// DE:
async function goTo(page) {
  await Auth.init();
  if (!Auth.isAuthenticated()) { window.location.href = '/auth/login?redirect=/' + page; return; }
  window.location.href = '/' + page;
}

// PARA:
async function goTo(page) {
  // Desabilitar todos os botões durante a verificação
  document.querySelectorAll('.home-btn').forEach(b => {
    b.style.opacity = '0.5';
    b.style.pointerEvents = 'none';
  });
  await Auth.init();
  if (!Auth.isAuthenticated()) {
    window.location.href = '/auth/login?redirect=/' + page;
    return;
  }
  window.location.href = '/' + page;
}
```
