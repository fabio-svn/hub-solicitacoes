## Contexto

2 ajustes no editor de templates:

1. Ícone da layer de "shape" no painel de layers e botão "+ Forma" deve 
   usar cor ruby red (#AC3631), igual aos outros ícones.

2. Botão "Voltar" às vezes não dispara navegação — goToList() chamada 
   manualmente funciona, mas via click falha silenciosamente. Causa 
   provável: unhandled promise rejection dentro do goToList chamado 
   sem await. Solução: adicionar error handling + fallback.

Arquivo: artifacts/api-server/public/admin-templates.html

## Fix 2 — Robustez do botão Voltar

LOCALIZAR a função `requestBack` (~linha 1014):
```js
function requestBack() {
  console.log('[back] requestBack, isDirty=', isDirty);
  if (!isDirty) { goToList(); return; }
  pendingNavAction = goToList;
  const modal = document.getElementById('unsavedModal');
  console.log('[back] modal element=', modal, 'classes=', modal?.className);
  modal.classList.add('open');
  console.log('[back] after add open, classes=', modal.className);
}
```

SUBSTITUIR por:
```js
function requestBack() {
  console.log('[back] click detected, isDirty=', isDirty);
  
  if (!isDirty) {
    // Chamar goToList com tratamento de erro — evita unhandled rejection silenciosa
    goToList()
      .then(() => console.log('[back] goToList ok'))
      .catch(err => {
        console.error('[back] goToList falhou — usando fallback:', err);
        // Fallback: força showView manualmente
        try {
          const ed = document.getElementById('view-editor');
          const ls = document.getElementById('view-list');
          if (ed) ed.style.display = 'none';
          if (ls) ls.style.display = 'flex';
        } catch (e2) {
          console.error('[back] fallback também falhou:', e2);
        }
      });
    return;
  }
  
  pendingNavAction = goToList;
  const modal = document.getElementById('unsavedModal');
  if (modal) {
    modal.classList.add('open');
  } else {
    console.warn('[back] modal não existe — usando confirm nativo');
    if (confirm('Você tem alterações não salvas. Sair sem salvar?')) {
      isDirty = false;
      goToList().catch(err => {
        console.error('[back] goToList falhou:', err);
        try {
          document.getElementById('view-editor').style.display = 'none';
          document.getElementById('view-list').style.display = 'flex';
        } catch {}
      });
    }
  }
}
```

E LOCALIZAR `goToList` (~linha 993):
```js
async function goToList() {
  await loadAllTemplates();
  if (createFormSchemas.length === 0) {
    try { const r = await fetch('/api/admin/form-schemas'); if (r.ok) createFormSchemas = await r.json(); } catch {}
  }
  renderListView();
  showView('list');
}
```

SUBSTITUIR por (com error handling em cada etapa, garantindo que showView 
sempre roda):
```js
async function goToList() {
  // Tenta carregar templates, mas se falhar não bloqueia saída do editor
  try {
    await loadAllTemplates();
  } catch (err) {
    console.error('[goToList] loadAllTemplates falhou:', err);
  }
  
  if (createFormSchemas.length === 0) {
    try {
      const r = await fetch('/api/admin/form-schemas');
      if (r.ok) createFormSchemas = await r.json();
    } catch {}
  }
  
  try {
    renderListView();
  } catch (err) {
    console.error('[goToList] renderListView falhou:', err);
  }
  
  // showView sempre roda (mesmo se loadAllTemplates ou renderListView falharem)
  // Garante que o user SAI do editor
  showView('list');
}
```

SUBSTITUIR por (mesmo tratamento de erro):
```js
function exitWithoutSave() {
  isDirty = false;
  closeUnsavedModal();
  if (pendingNavAction) {
    const action = pendingNavAction;
    pendingNavAction = null;
    const result = action();
    if (result && typeof result.catch === 'function') {
      result.catch(err => console.error('[exitWithoutSave] action falhou:', err));
    }
  }
}
```
