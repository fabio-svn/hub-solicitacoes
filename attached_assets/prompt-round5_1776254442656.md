## 3. DRAWER — Labels legíveis para materiais e valores internos

Na função `openDrawer()` do `dashboard.html`, adicionar um mapeamento de valores internos para labels legíveis antes de renderizar o drawer. Isso resolve "pacote-padrao" → "Pacote Padrão", "presencial" → "Presencial", etc.

Adicionar esta função no `<script>` do `dashboard.html`:

```js
function humanizeValue(key, value) {
  // Mapas de valores internos para labels legíveis
  const maps = {
    natureza: { presencial: 'Presencial', online: 'Online' },
    maturidade: { 1: 'Tenho a maioria das informações', 2: 'Tenho algumas informações', 3: 'Ainda estou estruturando' },
    localEvento: { unidade: 'Unidade SVN', externo: 'Local externo', 'nao-definido': 'Não definido' },
    publico: {
      'Fechado para convidados': 'Fechado para convidados',
      'Aberto ao publico em geral': 'Aberto ao público em geral',
    },
  };

  // Mapeamento de ids de materiais para labels
  const materiaisMap = {};
  if (typeof ITENS_MATERIAIS !== 'undefined') {
    ITENS_MATERIAIS.forEach(i => { materiaisMap[i.id] = i.label; });
  }
  if (typeof ITENS_MATERIAIS_ONLINE !== 'undefined') {
    ITENS_MATERIAIS_ONLINE.forEach(i => { materiaisMap[i.id] = i.label; });
  }

  // Arrays de materiais
  if (key === 'materiais' && Array.isArray(value)) {
    return value.map(v => materiaisMap[v] || v).join(', ');
  }

  // Arrays de selos
  if (key === 'selos' && Array.isArray(value)) {
    if (typeof SELOS_ASSESSOR !== 'undefined') {
      const selosMap = {};
      SELOS_ASSESSOR.forEach(s => { selosMap[s.id] = s.label; });
      return value.map(v => selosMap[v] || v).join(' · ');
    }
  }

  // Lookup por chave
  if (maps[key] && maps[key][value] !== undefined) {
    return maps[key][value];
  }

  // Valores genéricos com traço (ids internos como "pacote-padrao")
  if (typeof value === 'string' && value.includes('-') && !value.includes(' ')) {
    // Verificar se é um id de material
    if (materiaisMap[value]) return materiaisMap[value];
  }

  return value;
}
```

Na função `openDrawer()`, ao renderizar cada campo, envolver o valor em `humanizeValue()`:

```js
// Substituir:
<div class="drawer-field-value">${esc(String(value))}</div>

// Por:
<div class="drawer-field-value">${esc(String(humanizeValue(key, value)))}</div>
```

E para arrays simples:
```js
// Substituir:
<div class="drawer-field-value">${value.map(v => esc(String(v))).join(' · ')}</div>

// Por:
<div class="drawer-field-value">${esc(humanizeValue(key, value))}</div>
```

## 4. DRAWER — Corrigir lentidão e falha ao abrir

O drawer demora ou não abre porque a função `openDrawer()` faz um fetch para `/api/solicitacoes/:id/status` a cada clique — se o servidor demorar ou o endpoint falhar, o drawer fica travado aguardando.

### 4a. Tornar o fetch de status não-bloqueante

Substituir o início da função `openDrawer()` por:

```js
async function openDrawer(item) {
  // Abrir o drawer IMEDIATAMENTE, sem aguardar o fetch de status
  document.getElementById('drawerOverlay').classList.add('visible');
  document.getElementById('drawer').classList.add('open');

  // Renderizar com os dados que já temos em cache
  renderDrawerContent(item);

  // Atualizar status em background (sem bloquear a abertura)
  try {
    const syncRes = await fetch('/api/solicitacoes/' + item.id + '/status');
    if (syncRes.ok) {
      const syncData = await syncRes.json();
      if (syncData.updated && syncData.status !== item.status) {
        item.status = syncData.status;
        // Re-renderizar apenas o cabeçalho com o novo status
        renderDrawerContent(item);
      }
    }
  } catch {}
}
```

    dados = typeof item.dados === 'string' ? JSON.parse(item.dados) : (item.dados || {});
  } catch { dados = {}; }

        </div>`;

        <p style="font-size:0.7rem;font-weight:600;letter-spacing:0.07em;text-transform:uppercase;opacity:0.4;margin-bottom:14px">Andamento da solicitação</p>
        <div style="display:flex;align-items:center;gap:0;padding-bottom:28px">
          ${etapasVisiveis.map((etapa, idx) => {
            const isConcluida = idx < idxAtual;
            const isAtual = idx === idxAtual;
            const isUltima = idx === etapasVisiveis.length - 1;
            const cor = getStatusColor(etapa.id);

            const bg = isConcluida ? '#065f46' : isAtual ? cor.bg : 'transparent';
            const border = isConcluida ? '#065f46' : isAtual ? cor.text : 'rgba(34,27,25,0.15)';
            const icone = isConcluida
              ? `<svg width="9" height="9" viewBox="0 0 24 24" fill="none" stroke="white" stroke-width="3"><polyline points="20 6 9 17 4 12"/></svg>`
              : isAtual ? `<div style="width:6px;height:6px;border-radius:50%;background:${cor.text}"></div>` : '';
            const labelAtual = isAtual
              ? `<div style="position:absolute;top:28px;left:50%;transform:translateX(-50%);white-space:nowrap;font-size:0.68rem;font-weight:700;color:${cor.text}">${etapa.label}</div>`
              : '';

            return `
              <div style="display:flex;align-items:center;flex:${isUltima ? '0' : '1'};position:relative">

  // Campos do formulário
  let bodyHtml = '';
