// Em Auth.init(), após definir this.user, buscar pendências:

async init() {
  // ...código existente...

  // Após autenticar, buscar solicitações pendentes de aprovação:
  if (this.user) {
    try {
      const pendRes = await fetch('/api/solicitacoes/pendentes-aprovacao');
      if (pendRes.ok) {
        const pendData = await pendRes.json();
        this._pendenteIds = new Set(pendData.ids || []);
        // Filtrar os já lidos (localStorage)
        const lidos = Auth._getLidos();
        this._pendenteIds = new Set([...this._pendenteIds].filter(id => !lidos.has(id)));
      }
    } catch {}
  }

  return this.user;
},

// Método auxiliar — IDs lidos persistem no localStorage
_getLidos() {
  try {
    return new Set(JSON.parse(localStorage.getItem('svn_aprovacao_lidos') || '[]'));
  } catch { return new Set(); }
},

marcarComoLido(id) {
  try {
    const lidos = Auth._getLidos();
    lidos.add(id);
    localStorage.setItem('svn_aprovacao_lidos', JSON.stringify([...lidos]));
    if (this._pendenteIds) this._pendenteIds.delete(id);
    this._atualizarBadgeHeader();
  } catch {}
},

temPendencias() {
  return (this._pendenteIds?.size || 0) > 0;
},

_atualizarBadgeHeader() {
  const badge = document.getElementById('notifBadgeAvatar');
  const badgeMenu = document.getElementById('notifBadgeMenu');
  const tem = this.temPendencias();
  if (badge) badge.style.display = tem ? 'block' : 'none';
  if (badgeMenu) badgeMenu.style.display = tem ? 'inline-flex' : 'none';
},
```

async function loadSolicitacao() {
  try {
    const res = await fetch('/api/solicitacoes/' + solicitacaoId);
    if (!res.ok) { window.location.href = '/dashboard.html'; return; }
    const item = await res.json();
    renderPage(item);

      Auth.marcarComoLido(item.id);
    }
  } catch (e) {
    window.location.href = '/dashboard.html';
  }
}

// Funções de controle de seleção:

let _selecionados = new Set();

function toggleCheckAll(cb) {
  document.querySelectorAll('.row-check').forEach(c => {
    c.checked = cb.checked;
    const id = parseInt(c.dataset.id);
    if (cb.checked) _selecionados.add(id);
    else _selecionados.delete(id);
  });
  atualizarSelecao();
}

function atualizarSelecao() {
  _selecionados.clear();
  document.querySelectorAll('.row-check:checked').forEach(c => {
    _selecionados.add(parseInt(c.dataset.id));
  });
  const bar = document.getElementById('acoesMassaBar');
  const count = document.getElementById('acoesMassaCount');
  const n = _selecionados.size;
  if (n > 0) {
    bar.style.display = 'flex';
    count.textContent = n + ' solicitaç' + (n === 1 ? 'ão selecionada' : 'ões selecionadas');
  } else {
    bar.style.display = 'none';
  }
  // Atualizar checkAll
  const all = document.querySelectorAll('.row-check');
  const checked = document.querySelectorAll('.row-check:checked');
  const checkAll = document.getElementById('checkAll');
  if (checkAll) checkAll.indeterminate = checked.length > 0 && checked.length < all.length;
  if (checkAll) checkAll.checked = all.length > 0 && checked.length === all.length;
}

function limparSelecao() {
  _selecionados.clear();
  document.querySelectorAll('.row-check').forEach(c => c.checked = false);
  const checkAll = document.getElementById('checkAll');
  if (checkAll) { checkAll.checked = false; checkAll.indeterminate = false; }
  atualizarSelecao();
}

async function excluirSelecionados() {
  if (_selecionados.size === 0) return;
  const n = _selecionados.size;
  document.getElementById('deleteMassaModal').style.display = 'flex';
  document.getElementById('deleteMassaCount').textContent =
    n + ' solicitaç' + (n === 1 ? 'ão' : 'ões');
}

async function confirmarExclusaoMassa() {
  const btn = document.getElementById('btnConfirmarExclusaoMassa');
  btn.disabled = true;
  btn.textContent = 'Excluindo...';
  const ids = [..._selecionados];
  let erros = 0;
  for (const id of ids) {
    try {
      const res = await fetch('/api/solicitacoes/' + id, { method: 'DELETE' });
      if (!res.ok) erros++;
    } catch { erros++; }
  }
  document.getElementById('deleteMassaModal').style.display = 'none';
  btn.disabled = false;
  btn.textContent = 'Excluir permanentemente';
  limparSelecao();
  // Recarregar histórico
  adminPageHistorico = 1;
  loadAdminHistorico();
  if (erros > 0) alert(`${erros} solicitaç${erros === 1 ? 'ão' : 'ões'} não puderam ser excluídas.`);
}

function cancelarExclusaoMassa() {
  document.getElementById('deleteMassaModal').style.display = 'none';
}
```
