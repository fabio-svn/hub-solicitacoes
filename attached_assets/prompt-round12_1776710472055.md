```js
let _deleteTargetId = null;
let _deleteTargetRow = null;

function confirmarExcluir(id, btn) {
  _deleteTargetId = id;
  _deleteTargetRow = btn.closest('tr');
  const modal = document.getElementById('deleteModal');
  modal.style.display = 'flex';
}

function cancelarExclusao() {
  _deleteTargetId = null;
  _deleteTargetRow = null;
  document.getElementById('deleteModal').style.display = 'none';
}

async function executarExclusao() {
  if (!_deleteTargetId) return;
  const btn = document.getElementById('btnConfirmarExclusao');
  btn.disabled = true;
  btn.textContent = 'Excluindo...';

  try {
    const res = await fetch('/api/solicitacoes/' + _deleteTargetId, { method: 'DELETE' });
    if (res.ok) {
      document.getElementById('deleteModal').style.display = 'none';
      if (_deleteTargetRow) {
        _deleteTargetRow.style.transition = 'opacity 0.3s, transform 0.3s';
        _deleteTargetRow.style.opacity = '0';
        _deleteTargetRow.style.transform = 'translateX(20px)';
        setTimeout(() => { _deleteTargetRow?.remove(); }, 300);
      }
    } else {
      const data = await res.json().catch(() => ({}));
      btn.disabled = false;
      btn.textContent = 'Excluir permanentemente';
      alert(data.error || 'Erro ao excluir. Tente novamente.');
    }
  } catch {
    btn.disabled = false;
    btn.textContent = 'Excluir permanentemente';
    alert('Erro de conexão.');
  }
  _deleteTargetId = null;
  _deleteTargetRow = null;
}

// Fechar com Escape
document.addEventListener('keydown', e => {
  if (e.key === 'Escape') cancelarExclusao();
});
```

## 4. `dashboard.html` — Gráfico: corrigir filtro de checkboxes + dropdown

### 4a. O problema atual: os checkboxes ficam sempre marcados

O root cause é que `renderGraficoCheckboxes()` recria os elementos
toda vez que o gráfico é redesenhado, mas `graficoFiltrosAtivos` é
`null` (significa "todos ativos"), então os novos checkboxes sempre
aparecem como `checked`.

Solução: inicializar `graficoFiltrosAtivos` como Set completo ao
primeiro render, para que desmarcar um item realmente funcione:

```js
function renderAdminGrafico() {
  if (!adminGraficoData) return;

  const todosItens = adminGraficoModo === 'tipo'
    ? adminGraficoData.porTipo.map(i => ({
        id: i.tipo,
        label: (typeof TIPO_SOLICITACAO_LABELS !== 'undefined' && TIPO_SOLICITACAO_LABELS[i.tipo]) || i.tipo,
        value: i.count
      }))
    : adminGraficoData.porStatus.map(i => ({
        id: i.status,
        label: (STATUS_SOLICITACAO?.find(s => s.id === i.status)?.label) || i.status,
        value: i.count,
        cor: STATUS_SOLICITACAO?.find(s => s.id === i.status)?.bg || null,
      }));

  // Inicializar filtro com todos os itens na primeira vez
  if (graficoFiltrosAtivos === null) {
    graficoFiltrosAtivos = new Set(todosItens.map(i => i.id));
  }

  // Filtrar apenas os itens com IDs no Set
  const itens = todosItens.filter(i => graficoFiltrosAtivos.has(i.id));

  // Renderizar checkboxes — preserva estado atual
  renderGraficoCheckboxes(todosItens);

  // ... (resto do desenho do canvas com `itens` filtrados — igual ao existente)
}

function toggleGraficoFiltro(id) {
  if (!graficoFiltrosAtivos) return;
  if (graficoFiltrosAtivos.has(id)) {
    graficoFiltrosAtivos.delete(id);
  } else {
    graficoFiltrosAtivos.add(id);
  }
  renderAdminGrafico();
}

function resetGraficoFiltros() {
  graficoFiltrosAtivos = null; // força reinicialização com todos marcados
  renderAdminGrafico();
}

function switchGrafico(modo) {
  adminGraficoModo = modo;
  graficoFiltrosAtivos = null; // reset ao trocar modo
  document.getElementById('graficoBtnTipo').classList.toggle('active', modo === 'tipo');
  document.getElementById('graficoBtnStatus').classList.toggle('active', modo === 'status');
  renderAdminGrafico();
}
```

```js
function renderGraficoCheckboxes(todosItens) {
  const container = document.getElementById('graficoCheckboxes');
  if (!container) return;
  const CORES = ['#AC3631','#221B19','#4a7c2f','#1e40af','#9333ea','#ea580c','#0891b2','#be123c','#854d0e','#166534'];

  container.innerHTML = '';
  todosItens.forEach((item, i) => {
    const isAtivo = graficoFiltrosAtivos ? graficoFiltrosAtivos.has(item.id) : true;
    const cor = item.cor || CORES[i % CORES.length];

    const label = document.createElement('label');
    label.style.cssText = `display:inline-flex;align-items:center;gap:6px;padding:5px 12px;border-radius:999px;border:1.5px solid ${isAtivo ? cor : 'rgba(34,27,25,0.12)'};background:${isAtivo ? cor + '18' : 'transparent'};cursor:pointer;font-size:0.78rem;font-weight:600;transition:all 0.15s;user-select:none`;

    const dot = document.createElement('span');
    dot.style.cssText = `width:8px;height:8px;border-radius:2px;background:${cor};flex-shrink:0`;

    const text = document.createTextNode(item.label);

    label.appendChild(dot);
    label.appendChild(text);
    label.addEventListener('click', () => toggleGraficoFiltro(item.id));

    container.appendChild(label);
  });
}
```

## 5. `auth.js` — Refresh automático ao impersonar

### Substituir a função `window._impersonar`:

```js
window._impersonar = async function() {
  const input = document.getElementById('impersonarEmailInput');
  const email = input?.value?.trim();
  if (!email || !email.includes('@')) {
    if (input) input.style.borderColor = '#dc2626';
    return;
  }
  try {
    const res = await fetch('/api/admin/impersonate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email }),
    });
    if (res.ok) {
      sessionStorage.setItem('svn_impersonate', email);
      // Fechar dropdown se aberto
      const dropdown = document.querySelector('.user-dropdown');
      if (dropdown) dropdown.style.display = 'none';
      // Refresh imediato — garante que nenhum dado de admin fica visível
      window.location.reload();
    } else {
      alert('Não foi possível entrar como esse usuário.');
    }
  } catch { alert('Erro de conexão.'); }
};
```
