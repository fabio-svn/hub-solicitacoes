### 6.5 Upload de foto do palestrante — corrigir preview e adicionar feedback
Corrigir o handler de upload para atualizar o avatar corretamente e adicionar mensagem de status:

```js
// Para cada bloco de palestrante, usar este padrão:
fotoInput.addEventListener('change', function() {
  if (!this.files || !this.files[0]) return;
  const file = this.files[0];
  const reader = new FileReader();
  reader.onload = function(e) {
    // Atualizar o avatar com a foto
    const avatarEl = document.getElementById('speakerAvatar_' + speakerIndex);
    if (avatarEl) {
      avatarEl.innerHTML = `<img src="${e.target.result}" alt="foto" style="width:100%;height:100%;object-fit:cover;border-radius:50%">`;
    }
  };
  reader.onerror = function() {
    statusEl.textContent = '✗ Erro ao carregar foto. Tente novamente.';
    statusEl.style.color = 'var(--ruby-red)';
  };
  reader.readAsDataURL(file);
  // Feedback imediato
  const statusEl = document.getElementById('speakerPhotoStatus_' + speakerIndex);
  if (statusEl) {
    statusEl.textContent = '✓ Foto carregada com sucesso';
    statusEl.style.color = 'var(--sage-green)';
    statusEl.style.fontSize = '0.8rem';
    statusEl.style.marginTop = '4px';
    statusEl.style.display = 'block';
  }
});
```

Adicionar um elemento `<span id="speakerPhotoStatus_N" style="display:none"></span>` abaixo de cada input de foto de palestrante.
