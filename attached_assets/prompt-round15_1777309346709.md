const MARCAS_CONTRATO = ['SVN Capital', 'SVN Connect', 'SVN Investimentos'];

function handleMarcaChange() {
  const marca = document.getElementById('marca').value;
  const isContrato = MARCAS_CONTRATO.includes(marca);

      .forEach(r => r.checked = false);
  }

### 10b. Criar função utilitária de envio de webhook:

```ts
// Adicionar no topo de forms.ts, após os imports:
async function dispararWebhook(
  webhookUrl: string,
  fields: Record<string, string>,
  meta: { post_id?: string; form_id?: string } = {}
): Promise<void> {
  const body = new URLSearchParams();
  Object.entries(fields).forEach(([k, v]) => {
    body.append(`form_fields[${k}]`, v);
  });
  if (meta.post_id) body.append('post_id', meta.post_id);
  if (meta.form_id) body.append('form_id', meta.form_id);
  body.append('action', 'elementor_pro_forms_send_form');

  await fetch(webhookUrl, {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: body.toString(),
  });
}
```
