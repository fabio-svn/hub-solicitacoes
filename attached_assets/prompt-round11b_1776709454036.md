## 1. `forms.ts` — Endpoint DELETE `/solicitacoes/:id` (apenas admins)

```ts
router.delete("/solicitacoes/:id", requireAuth, async (req, res): Promise<void> => {
  try {
    const user = req.session.user!;
    if (user.role !== "admin" && user.role !== "gestor") {
      res.status(403).json({ error: "Acesso negado" }); return;
    }

    const id = parseInt(String(req.params.id));
    if (isNaN(id)) { res.status(400).json({ error: "ID inválido" }); return; }

    const [solicitacao] = await db.select()
      .from(solicitacoesTable)
      .where(eq(solicitacoesTable.id, id));

    if (!solicitacao) { res.status(404).json({ error: "Solicitação não encontrada" }); return; }

    // Deletar arquivos vinculados primeiro
    await db.delete(arquivosTable).where(eq(arquivosTable.solicitacao_id, id));

    // Deletar solicitação
    await db.delete(solicitacoesTable).where(eq(solicitacoesTable.id, id));

    logger.info({ id, deletedBy: user.email }, "Solicitação excluída por admin");
    res.json({ success: true });
  } catch (err) {
    logger.error({ err }, "Erro ao excluir solicitação");
    res.status(500).json({ error: "Erro ao excluir solicitação" });
  }
});
```

### 2d. Adicionar função `confirmarExcluir()`

```js
async function confirmarExcluir(id, btn) {
  // Confirmação inline no botão em vez de alert()
  if (btn.dataset.confirmando === '1') {
    // Segunda vez — executar
    btn.dataset.confirmando = '0';
    btn.title = 'Excluir solicitação';
    btn.innerHTML = `<svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="3 6 5 6 21 6"/><path d="M19 6l-1 14a2 2 0 01-2 2H8a2 2 0 01-2-2L5 6"/><path d="M10 11v6"/><path d="M14 11v6"/><path d="M9 6V4a1 1 0 011-1h4a1 1 0 011 1v2"/></svg>`;
    btn.style.color = '';

    try {
      const res = await fetch('/api/solicitacoes/' + id, { method: 'DELETE' });
      if (res.ok) {
        // Remover linha com animação
        const row = btn.closest('tr');
        if (row) {
          row.style.transition = 'opacity 0.3s, transform 0.3s';
          row.style.opacity = '0';
          row.style.transform = 'translateX(20px)';
          setTimeout(() => { row.remove(); }, 300);
        }
      } else {
        const data = await res.json();
        alert(data.error || 'Erro ao excluir.');
      }
    } catch { alert('Erro de conexão.'); }
    return;
  }

  // Primeira vez — pedir confirmação via estilo do botão
  btn.dataset.confirmando = '1';
  btn.style.color = '#dc2626';
  btn.title = 'Clique novamente para confirmar';
  btn.innerHTML = `<svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="20 6 9 17 4 12"/></svg>`;

  // Cancelar após 3 segundos se não confirmar
  setTimeout(() => {
    if (btn.dataset.confirmando === '1') {
      btn.dataset.confirmando = '0';
      btn.style.color = '';
      btn.title = 'Excluir solicitação';
      btn.innerHTML = `<svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="3 6 5 6 21 6"/><path d="M19 6l-1 14a2 2 0 01-2 2H8a2 2 0 01-2-2L5 6"/><path d="M10 11v6"/><path d="M14 11v6"/><path d="M9 6V4a1 1 0 011-1h4a1 1 0 011 1v2"/></svg>`;
    }
  }, 3000);
}
```

## OBSERVAÇÕES

- `forms.ts` precisa de build após adicionar o endpoint DELETE
- `dashboard.html` não precisa de build
- A confirmação de exclusão é inline (double-click pattern): primeiro
  clique muda o ícone para ✓ e cor vermelha, segundo clique executa,
  3 segundos sem confirmar cancela automaticamente — sem popup do sistema
- A classe `.admin-tabela` é necessária para os seletores CSS do sticky
- O `min-width:700px` na tabela garante que o scroll horizontal
  aparece em telas menores sem quebrar o layout
