### 3b. Implementar função de validação global

Adicionar em **todos os formulários** (`form-eventos.html`, `form-pagina-assessores.html`, `form-apresentacoes.html`, `form-artes-divulgacao.html`, `form-atualizacao-material.html`, `form-criacao-pdf.html`) a seguinte função de validação:

```js
function validateRequiredFields() {
  // Remover estado de erro anterior
  document.querySelectorAll('.field-invalid').forEach(el => {
    el.classList.remove('field-invalid');
  });

  const invalidFields = [];

  // Verificar todos os campos com atributo required visíveis
  document.querySelectorAll('input[required], select[required], textarea[required]').forEach(input => {
    // Ignorar campos ocultos (dentro de steps inativos ou campos condicionais fechados)
    const step = input.closest('.form-step');
    if (step && !step.classList.contains('active')) return;
    const conditional = input.closest('.conditional-fields');
    if (conditional && !conditional.classList.contains('open')) return;

    const isEmpty = input.type === 'radio' || input.type === 'checkbox'
      ? !document.querySelector(`input[name="${input.name}"]:checked`)
      : !input.value.trim();

    if (isEmpty) {
      const fieldWrapper = input.closest('.field');
      if (fieldWrapper) {
        fieldWrapper.classList.add('field-invalid');
        // Adicionar mensagem de erro se não houver uma
        let errorEl = fieldWrapper.querySelector('.field-error');
        if (!errorEl) {
          errorEl = document.createElement('div');
          errorEl.className = 'field-error';
          input.parentElement.appendChild(errorEl);
        }
        errorEl.textContent = 'Este campo é obrigatório.';
        invalidFields.push(fieldWrapper);
      }
    }
  });

  if (invalidFields.length > 0) {
    // Rolar até o primeiro campo inválido
    invalidFields[0].scrollIntoView({ behavior: 'smooth', block: 'center' });
    return false;
  }

  return true;
}
```

Quando um campo condicional abrir (`.conditional-fields.open`), os inputs dentro dele devem receber `required` dinamicamente. Quando fechar, remover `required`:

```js
function toggleConditionalField(el, fieldId) {
  const fields = document.getElementById(fieldId);
  if (!fields) return;
  const isOpen = fields.classList.contains('open');
  fields.classList.toggle('open', !isOpen);
  // Tornar campos obrigatórios/opcionais dinamicamente
  fields.querySelectorAll('input, select, textarea').forEach(input => {
    const isOptional = input.dataset.optional === 'true';
    if (!isOptional) {
      input.required = !isOpen; // se estava aberto e vai fechar, remover required
    }
  });
}
```

Para campos que devem permanecer opcionais mesmo quando visíveis, adicionar `data-optional="true"` no input.

### 16e. Data relativa nos cards

Adicionar função de data relativa no `dashboard.html`:

```js
function dataRelativa(dateStr) {
  const data = new Date(dateStr);
  const hoje = new Date();
  const diff = Math.floor((hoje - data) / (1000 * 60 * 60 * 24));
  if (diff === 0) return 'Hoje';
  if (diff === 1) return 'Ontem';
  if (diff < 7) return `Há ${diff} dias`;
  if (diff < 30) return `Há ${Math.floor(diff / 7)} semana${Math.floor(diff / 7) > 1 ? 's' : ''}`;
  return data.toLocaleDateString('pt-BR');
}
```

Substituir no `renderList()`:
```js
// De:

// Para:
const date = dataRelativa(item.created_at);
```

### 16f. Agrupamento por período na lista

Na função `renderList()`, após ordenar os dados, agrupar por período antes de renderizar:

```js
function agruparPorPeriodo(items) {
  const grupos = { 'Esta semana': [], 'Este mês': [], 'Anteriores': [] };
  const hoje = new Date();
  items.forEach(item => {
    const data = new Date(item.created_at);
    const diff = Math.floor((hoje - data) / (1000 * 60 * 60 * 24));
    if (diff <= 7) grupos['Esta semana'].push(item);
    else if (diff <= 30) grupos['Este mês'].push(item);
    else grupos['Anteriores'].push(item);
  });
  return grupos;
}
```
