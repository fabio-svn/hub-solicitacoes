Função checkAlignmentGuides:

js   const ALIGN_THRESHOLD = 5; // px
   
   function checkAlignmentGuides(el) {
     const canvas = document.getElementById('canvasContainer'); // ou equivalente
     const canvasRect = canvas.getBoundingClientRect();
     const elRect = el.getBoundingClientRect();
     
     const elCenterX = elRect.left + elRect.width / 2 - canvasRect.left;
     const elCenterY = elRect.top + elRect.height / 2 - canvasRect.top;
     const canvasCenterX = canvasRect.width / 2;
     const canvasCenterY = canvasRect.height / 2;
     
     const guideV = document.getElementById('alignGuideV');
     const guideH = document.getElementById('alignGuideH');
     
     // Centro horizontal (vertical line)
     if (Math.abs(elCenterX - canvasCenterX) < ALIGN_THRESHOLD) {
       guideV.style.left = canvasCenterX + 'px';
       guideV.style.display = 'block';
     } else {
       guideV.style.display = 'none';
     }
     
     // Centro vertical (horizontal line)
     if (Math.abs(elCenterY - canvasCenterY) < ALIGN_THRESHOLD) {
       guideH.style.top = canvasCenterY + 'px';
       guideH.style.display = 'block';
     } else {
       guideH.style.display = 'none';
     }
   }
   
   function hideAlignmentGuides() {
     document.getElementById('alignGuideH').style.display = 'none';
     document.getElementById('alignGuideV').style.display = 'none';
   }

jsfunction alignSelectedTo(direction) {
  if (!selectedLayerId || !currentTemplate) return;
  const layer = getLayer(selectedLayerId);
  if (!layer) return;
  
  const canvasW = currentTemplate.canvas?.width || 1080;
  const canvasH = currentTemplate.canvas?.height || 1080;
  const layerW = layer.w || 100;
  const layerH = layer.h || (layer.type === 'text-line' ? 60 : 100);
  
  switch(direction) {
    case 'left':     layer.x = 0; break;
    case 'center-h': layer.x = Math.round((canvasW - layerW) / 2); break;
    case 'right':    layer.x = canvasW - layerW; break;
    case 'top':      layer.y = 0; break;
    case 'center-v': layer.y = Math.round((canvasH - layerH) / 2); break;
    case 'bottom':   layer.y = canvasH - layerH; break;
  }
  
  markDirty();
  pushHistory();
  renderCanvas();      // re-render do canvas — adaptar nome se for outro
  showProps(selectedLayerId);  // atualiza UI
  scheduleLivePreview(500);
}
