  <script>
    const TIPO = 'ch-kit-onboarding';

    async function init() {
      try { if (typeof _configReady !== 'undefined') await _configReady; } catch (e) {}
      await Auth.init();
      if (!Auth.isAuthenticated()) { window.location.href = '/auth/login?redirect=/form-ch-kit-onboarding.html'; return; }
      const role = Auth.getUserRole();
      if (role !== 'capital_humano' && role !== 'admin') { window.location.href = '/dashboard.html'; return; }
      if (typeof Shell !== 'undefined' && Shell.render) {
        Shell.render({ activeRoute: 'capital-humano', contentEl: document.getElementById('pageContent') });
      }
      const sel = document.getElementById('unidade');
      const unidades = (typeof UNIDADES_SVN !== 'undefined' && Array.isArray(UNIDADES_SVN)) ? UNIDADES_SVN : [];
      unidades.forEach(function (u) {
        const nome = typeof u === 'string' ? u : (u.nome || u.label || u.value || '');
        if (!nome) return;
        const o = document.createElement('option'); o.value = nome; o.textContent = nome; sel.appendChild(o);
      });
    }

    function validateRequiredFields() {
      document.querySelectorAll('.field-invalid').forEach(function (el) { el.classList.remove('field-invalid'); });
      const invalidFields = [];
      document.querySelectorAll('input[required], select[required], textarea[required]').forEach(function (input) {
        if (!input.value.trim()) {
          const fw = input.closest('.field');
          if (fw) {
            fw.classList.add('field-invalid');
            let e = fw.querySelector('.field-error');
            if (!e) { e = document.createElement('div'); e.className = 'field-error'; input.parentElement.appendChild(e); }
            e.textContent = 'Este campo é obrigatório.';
            invalidFields.push(fw);
          }
        }
      });
      if (invalidFields.length > 0) { invalidFields[0].scrollIntoView({ behavior: 'smooth', block: 'center' }); return false; }
      return true;
    }

    async function submitForm() {
      if (!validateRequiredFields()) return;
      const btn = document.getElementById('btnSubmit');
      btn.disabled = true; btn.textContent = 'Enviando...'; if (window.formLockUI) window.formLockUI(btn);

      const dados = {
        nome: Auth.getUserName(),
        unidade: document.getElementById('unidade').value,
        quantidade: document.getElementById('quantidade').value,
        dataEntrega: document.getElementById('dataEntrega').value,
      };

      const formData = new FormData();
      formData.append('tipo_solicitacao', TIPO);
      formData.append('dados', JSON.stringify(dados));

      try {
        const res = await fetch('/api/solicitacoes', { method: 'POST', body: formData });
        const d = await res.json().catch(function () { return {}; });
        if (res.ok) {
          try {
            sessionStorage.setItem('svn_ultimo_resumo', JSON.stringify({
              tipo: 'Kit Onboarding', solicitante: Auth.getUserName(),
              data: new Date().toLocaleDateString('pt-BR'),
              hora: new Date().toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' }),
              id: d.id,
            }));
          } catch (e) {}
          window.location.replace('/thankyou.html');
        } else {
          const errEl = document.getElementById('submitError');
          errEl.style.display = 'block'; errEl.textContent = d.error || 'Erro ao enviar. Tente novamente.';
          if (window.formUnlockUI) window.formUnlockUI(btn); else { btn.disabled = false; btn.textContent = 'Enviar'; }
        }
      } catch (err) {
        const errEl = document.getElementById('submitError');
        errEl.style.display = 'block'; errEl.textContent = 'Erro de rede. Tente novamente.';
        if (window.formUnlockUI) window.formUnlockUI(btn); else { btn.disabled = false; btn.textContent = 'Enviar'; }
      }
    }

    init();
  </script>
