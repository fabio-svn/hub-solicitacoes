### 3b. Implementar função de validação global

Adicionar em **todos os formulários** (`form-eventos.html`, `form-pagina-assessores.html`, `form-apresentacoes.html`, `form-artes-divulgacao.html`, `form-atualizacao-material.html`, `form-criacao-pdf.html`) a seguinte função de validação:

```js
function validateRequiredFields() {
  // Remover estado de erro anterior
  document.querySelectorAll('.field-invalid').forEach(el => {
    el.classList.remove('field-invalid');
  });

  const invalidFields = [];

  // Verificar todos os campos com atributo required visíveis
  document.querySelectorAll('input[required], select[required], textarea[required]').forEach(input => {
    // Ignorar campos ocultos (dentro de steps inativos ou campos condicionais fechados)
    const step = input.closest('.form-step');
    if (step && !step.classList.contains('active')) return;
    const conditional = input.closest('.conditional-fields');
    if (conditional && !conditional.classList.contains('open')) return;

    const isEmpty = input.type === 'radio' || input.type === 'checkbox'
      ? !document.querySelector(`input[name="${input.name}"]:checked`)
      : !input.value.trim();

    if (isEmpty) {
      const fieldWrapper = input.closest('.field');
      if (fieldWrapper) {
        fieldWrapper.classList.add('field-invalid');
        // Adicionar mensagem de erro se não houver uma
        let errorEl = fieldWrapper.querySelector('.field-error');
        if (!errorEl) {
          errorEl = document.createElement('div');
          errorEl.className = 'field-error';
          input.parentElement.appendChild(errorEl);
        }
        errorEl.textContent = 'Este campo é obrigatório.';
        invalidFields.push(fieldWrapper);
      }
    }
  });

  if (invalidFields.length > 0) {
    // Rolar até o primeiro campo inválido
    invalidFields[0].scrollIntoView({ behavior: 'smooth', block: 'center' });
    return false;
  }

  return true;
}
```

### 3d. Limpar estado de erro ao usuário preencher o campo

Em cada formulário, adicionar listener para limpar o erro ao digitar:

```js
// Adicionar no init() de cada formulário, após renderizar o form:
document.addEventListener('input', function(e) {
  const field = e.target.closest('.field');
  if (field && field.classList.contains('field-invalid')) {
    if (e.target.value.trim()) {
      field.classList.remove('field-invalid');
    }
  }
});

document.addEventListener('change', function(e) {
  const field = e.target.closest('.field');
  if (field && field.classList.contains('field-invalid')) {
    field.classList.remove('field-invalid');
  }
});
```

## OBSERVAÇÕES

- A classe `.field-invalid` é adicionada apenas no submit — não durante o preenchimento normal
- O shake sutil chama atenção sem ser agressivo
- Campos dentro de etapas inativas (`form-step` sem `.active`) são ignorados na validação
- Campos dentro de campos condicionais fechados (`.conditional-fields` sem `.open`) também são ignorados
- Ao preencher um campo inválido, o estado de erro é removido imediatamente via listener de `input`/`change`
