### 3b. Adicionar endpoint `POST /admin/impersonate` em `forms.ts`

```ts
router.post("/admin/impersonate", requireAuth, async (req, res): Promise<void> => {
  try {
    const user = req.session.user!;
    if (user.role !== "admin" && user.role !== "gestor") {
      res.status(403).json({ error: "Acesso negado" }); return;
    }

    const { email } = req.body as { email: string };
    if (!email || !email.includes("@")) {
      res.status(400).json({ error: "E-mail inválido" }); return;
    }

    // Guardar usuário admin original na sessão
    req.session.adminOriginal = req.session.user;
    req.session.user = { ...req.session.user, email, name: email.split("@")[0], role: "user" };

    res.json({ success: true, email });
  } catch (err) {
    res.status(500).json({ error: "Erro ao impersonar" });
  }
});

router.post("/admin/impersonate/stop", requireAuth, async (req, res): Promise<void> => {
  try {
    if (req.session.adminOriginal) {
      req.session.user = req.session.adminOriginal;
      delete req.session.adminOriginal;
    }
    res.json({ success: true });
  } catch (err) {
    res.status(500).json({ error: "Erro ao sair" });
  }
});
```

### 3d. Funções JS de impersonação no `dashboard.html`

```js
async function impersonar() {
  const email = document.getElementById('impersonarEmail')?.value?.trim();
  if (!email || !email.includes('@')) {
    document.getElementById('impersonarEmail').style.borderColor = 'var(--ruby-red)';
    return;
  }

  try {
    const res = await fetch('/api/admin/impersonate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email }),
    });
    if (res.ok) {
      // Redirecionar para dashboard como o usuário impersonado
      window.location.href = '/dashboard.html?impersonate=' + encodeURIComponent(email);
    } else {
      alert('Não foi possível entrar como esse usuário.');
    }
  } catch {}
}

// No init(), verificar se está em modo impersonação
async function initImpersonarBanner() {
  const params = new URLSearchParams(window.location.search);
  const imp = params.get('impersonate');
  if (imp) {
    const banner = document.getElementById('impersonarBanner');
    const label = document.getElementById('impersonarEmailLabel');
    if (banner && label) {
      label.textContent = imp;
      banner.style.display = 'flex';
      // Adicionar padding-top ao body para não esconder conteúdo
      document.body.style.paddingTop = '44px';
    }
  }
}

async function sairImpersonar() {
  try {
    await fetch('/api/admin/impersonate/stop', { method: 'POST' });
```

### 5e. Substituir `toggleCampo()` por `toggleCampoInline()`

```js
const camposSelecionados = new Set();

function toggleCampoInline(btn) {
  const campo = btn.dataset.campo;
  const fieldEl = document.getElementById('field_' + campo);

  if (camposSelecionados.has(campo)) {
    // Fechar campo
    camposSelecionados.delete(campo);
    btn.classList.remove('selecionado');
    if (fieldEl) {
      fieldEl.style.maxHeight = '0';
      fieldEl.style.overflow = 'hidden';
      setTimeout(() => { fieldEl.style.display = 'none'; }, 300);
      // Remover required
      fieldEl.querySelectorAll('input,textarea,select').forEach(i => i.required = false);
    }
  } else {
    // Abrir campo
    camposSelecionados.add(campo);
    btn.classList.add('selecionado');
    if (fieldEl) {
      fieldEl.style.display = 'block';
      fieldEl.style.overflow = 'hidden';
      fieldEl.style.maxHeight = '0';
      // Transição de abertura
      requestAnimationFrame(() => {
        fieldEl.style.transition = 'max-height 0.3s ease';
        fieldEl.style.maxHeight = fieldEl.scrollHeight + 'px';
        setTimeout(() => { fieldEl.style.overflow = 'visible'; }, 300);
      });
      // Adicionar required (exceto opcionais)
      const opcionais = ['instagram', 'selos', 'depoimentos'];
      if (!opcionais.includes(campo)) {
        fieldEl.querySelectorAll('input,textarea,select').forEach(i => i.required = true);
      }
      // Scroll suave para o campo
      setTimeout(() => fieldEl.scrollIntoView({ behavior: 'smooth', block: 'nearest' }), 100);
    }
  }
  document.getElementById('seletorInlineError').style.display = 'none';
}
```
