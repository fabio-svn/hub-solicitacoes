## 1. `dashboard.html` — Fix bug botão "Excluindo..." travado

### Root cause
`btn.disabled = true` e `btn.textContent = 'Excluindo...'` são definidos
antes do fetch. Após sucesso, o modal fecha mas o botão nunca é resetado.
Na próxima abertura, o botão já está desabilitado e com texto errado.

### Corrigir `executarExclusao()` — resetar botão em todos os caminhos

```js
async function executarExclusao() {
  if (!_deleteTargetId) return;
  const btn = document.getElementById('btnConfirmarExclusao');
  btn.disabled = true;
  btn.textContent = 'Excluindo...';

  try {
    const res = await fetch('/api/solicitacoes/' + _deleteTargetId, { method: 'DELETE' });
    if (res.ok) {
      // Fechar modal e resetar botão ANTES da animação
      document.getElementById('deleteModal').style.display = 'none';
      btn.disabled = false;                        // ← ADICIONAR
      btn.textContent = 'Excluir permanentemente'; // ← ADICIONAR

      if (_deleteTargetRow) {
        const row = _deleteTargetRow;
        const height = row.getBoundingClientRect().height;
        row.style.transition = 'opacity 0.25s ease, transform 0.25s ease';
        row.style.opacity = '0';
        row.style.transform = 'translateX(24px)';
        setTimeout(() => {
          row.style.transition = 'all 0.2s ease';
          row.style.height = height + 'px';
          row.style.overflow = 'hidden';
          requestAnimationFrame(() => {
            row.style.height = '0';
            row.querySelectorAll('td').forEach(td => {
              td.style.padding = '0';
              td.style.border = 'none';
            });
          });
          setTimeout(() => row.remove(), 220);
        }, 260);
      }
    } else {
      const data = await res.json().catch(() => ({}));
      btn.disabled = false;                        // ← já existia
      btn.textContent = 'Excluir permanentemente'; // ← já existia
      alert(data.error || 'Erro ao excluir. Tente novamente.');
    }
  } catch {
    btn.disabled = false;
    btn.textContent = 'Excluir permanentemente';
    alert('Erro de conexão.');
  }
  _deleteTargetId = null;
  _deleteTargetRow = null;
}
```

### Também resetar ao fechar o modal (cancelarExclusao e Escape)

```js
function cancelarExclusao() {
  _deleteTargetId = null;
  _deleteTargetRow = null;
  document.getElementById('deleteModal').style.display = 'none';
  // Garantir reset do botão mesmo se fechou no meio de um processo
  const btn = document.getElementById('btnConfirmarExclusao');
  if (btn) {
    btn.disabled = false;
    btn.textContent = 'Excluir permanentemente';
  }
}
```

### 2c. Adicionar funções JS em `solicitacao.html`

```js
let _deleteTargetId = null;

function abrirModalExcluir(id) {
  _deleteTargetId = id;
  const modal = document.getElementById('deleteModal');
  modal.style.display = 'flex';
}

function cancelarExcluir() {
  _deleteTargetId = null;
  document.getElementById('deleteModal').style.display = 'none';
  const btn = document.getElementById('btnConfirmarExcluir');
  if (btn) { btn.disabled = false; btn.textContent = 'Excluir permanentemente'; }
}

async function confirmarExcluir() {
  if (!_deleteTargetId) return;
  const btn = document.getElementById('btnConfirmarExcluir');
  btn.disabled = true;
  btn.textContent = 'Excluindo...';

  try {
    const res = await fetch('/api/solicitacoes/' + _deleteTargetId, { method: 'DELETE' });
    if (res.ok) {
      // Redirecionar para o dashboard após excluir
      window.location.href = '/dashboard.html';
    } else {
      const data = await res.json().catch(() => ({}));
      btn.disabled = false;
      btn.textContent = 'Excluir permanentemente';
      alert(data.error || 'Erro ao excluir. Tente novamente.');
    }
  } catch {
    btn.disabled = false;
    btn.textContent = 'Excluir permanentemente';
    alert('Erro de conexão.');
  }
  _deleteTargetId = null;
}

// Fechar com Escape
document.addEventListener('keydown', e => {
  if (e.key === 'Escape') {
    cancelarExcluir();
    // Manter o fechar do modal de avaliação que já existe
    document.getElementById('avaliacaoModal').style.display = 'none';
  }
});
```

**Nota:** o listener de `keydown` para Escape já existe em `solicitacao.html`
fechando o `avaliacaoModal`. Substituir esse listener pelo novo acima que
fecha ambos os modais, evitando duplicação.

## OBSERVAÇÕES

- `dashboard.html` e `solicitacao.html` — nenhum build necessário
- O nome das funções em `solicitacao.html` é ligeiramente diferente
  (`abrirModalExcluir`, `cancelarExcluir`, `confirmarExcluir`) para não
  conflitar com as funções homônimas do `dashboard.html` caso algum script
  seja compartilhado no futuro
- Após exclusão em `solicitacao.html`, redireciona para `/dashboard.html`
  com um pequeno delay natural do redirecionamento — não precisa de animação
