### 10b. Adicionar em `auth.js` ou em um novo arquivo `transitions.js` (importar em todos os HTMLs)

```js
// Fade out ao clicar em links internos
document.addEventListener('click', function(e) {
  const link = e.target.closest('a[href]');
  if (!link) return;
  const href = link.getAttribute('href');
  // Só animar links internos (não âncoras, não externos, não #)
  if (!href || href.startsWith('#') || href.startsWith('http') || href.startsWith('mailto') || link.target === '_blank') return;
  e.preventDefault();
  document.body.classList.add('page-leaving');
  setTimeout(() => { window.location.href = href; }, 200);
});

// Tratar botão "Voltar" do navegador (pageshow com cache)
window.addEventListener('pageshow', function(e) {
  if (e.persisted) {
    // Página restaurada do bfcache — garantir que esteja visível
    document.body.classList.remove('page-leaving');
    document.body.style.opacity = '1';
  }
});
```

```js
function goStep(step) {
  saveFormState();
  const currentStep = document.querySelector('.form-step.active');
  if (currentStep) {
    currentStep.classList.add('leaving');

function activateStep(step) {

  const steps = getStepCount();

  document.getElementById('stepLabel').textContent = `Etapa ${step} de ${steps}`;

}
```
